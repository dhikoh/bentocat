    <!-- JavaScript for Smooth Aroma Transition -->
    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const dots = document.querySelectorAll('.dot');
            const aromaDisplay = document.getElementById('aroma-display');
            const aromaText = document.getElementById('aroma-text');

            dots.forEach(dot => {
                dot.addEventListener('click', function() {
                    // Hapus class active dari semua dot
                    dots.forEach(d => d.classList.remove('active'));
                    
                    // Tambahkan active ke dot yang di klik
                    this.classList.add('active');

                    const newAroma = this.getAttribute('data-aroma');
                    const newColor = this.getAttribute('data-color');

                    // Cegah animasi ulang jika mengklik aroma yang sama
                    if(aromaText.innerText === newAroma) return;

                    // 1. Mulai transisi efek liquid (huruf melebur, blur, dan menyusut)
                    aromaText.classList.add('liquid-hidden');

                    // 2. Tunggu saat teks sudah melebur sempurna, lalu ubah isinya
                    setTimeout(() => {
                        aromaText.innerText = newAroma;
                        aromaText.style.color = newColor;
                        aromaText.style.textShadow = `0 5px 25px ${newColor}80`;

                        // 3. Munculkan kembali cairan membentuk teks baru
                        aromaText.classList.remove('liquid-hidden');
                    }, 400); // 400ms menyesuaikan durasi animasi CSS
                });
            });
        });
    </script>
