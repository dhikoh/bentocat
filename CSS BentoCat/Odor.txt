    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const container = document.getElementById('molecule-system');
            const totalMolecules = 40; // Jumlah molekul bau

            for(let i = 0; i < totalMolecules; i++) {
                let molecule = document.createElement('div');
                molecule.className = 'odor-molecule';
                
                // Ukuran random (antara 6px - 14px)
                let size = (Math.random() * 8 + 6).toFixed(1) + 'px';
                
                // --- Kalkulasi Arah 360 Derajat ---
                // Sudut acak dari 0 hingga 360 derajat (dalam radian)
                let angle = Math.random() * Math.PI * 2;
                
                // Jarak awal dari pusat (Radius 60% - 80% memastikan muncul dari luar tengah)
                let startRadius = 60 + Math.random() * 20; 
                let startX = 50 + startRadius * Math.cos(angle);
                let startY = 50 + startRadius * Math.sin(angle);
                
                // Posisi tengah (mengayun natural sebelum ditarik paksa ke tengah)
                // Radius mengecil (30% - 40%), sudut bergeser sedikit agar gerakannya melengkung/swirl
                let midAngle = angle + (Math.random() * 0.8 - 0.4);
                let midRadius = 30 + Math.random() * 10;
                let midX = 50 + midRadius * Math.cos(midAngle);
                let midY = 50 + midRadius * Math.sin(midAngle);

                // Randomisasi timing animasi agar organik
                let delay = (Math.random() * 5).toFixed(2); // Delay 0 - 5 detik
                let duration = (4 + Math.random() * 3).toFixed(2); // Durasi 4 - 7 detik
                
                // Terapkan variabel CSS
                molecule.style.setProperty('--size', size);
                molecule.style.setProperty('--startX', `${startX.toFixed(2)}%`);
                molecule.style.setProperty('--startY', `${startY.toFixed(2)}%`);
                molecule.style.setProperty('--midX', `${midX.toFixed(2)}%`);
                molecule.style.setProperty('--midY', `${midY.toFixed(2)}%`);
                // --pullX dan --pullY dihapus karena kini menggunakan kursor (--mouseX/Y)
                molecule.style.setProperty('--delay', `${delay}s`);
                molecule.style.setProperty('--duration', `${duration}s`);
                
                container.appendChild(molecule);
            }

            // --- Logika Interaksi Kursor & Sentuhan (Sistem Sedot) ---
            const wrapper = document.querySelector('.illustration-wrapper');
            let targetX = 50, targetY = 50;
            let currentX = 50, currentY = 50;

            const updateMousePosition = (e) => {
                let clientX, clientY;
                if (e.touches && e.touches.length > 0) {
                    clientX = e.touches[0].clientX;
                    clientY = e.touches[0].clientY;
                } else {
                    clientX = e.clientX;
                    clientY = e.clientY;
                }
                
                const rect = wrapper.getBoundingClientRect();
                targetX = ((clientX - rect.left) / rect.width) * 100;
                targetY = ((clientY - rect.top) / rect.height) * 100;

                // Membatasi agar inti karbon tidak keluar dari kotak terlalu jauh
                targetX = Math.max(10, Math.min(90, targetX));
                targetY = Math.max(10, Math.min(90, targetY));
            };

            const resetMousePosition = () => {
                targetX = 50;
                targetY = 50;
            };

            // Event Listeners
            wrapper.addEventListener('mousemove', updateMousePosition);
            wrapper.addEventListener('touchmove', (e) => {
                e.preventDefault(); // Mencegah layar ter-scroll saat menggeser di dalam canvas
                updateMousePosition(e);
            }, { passive: false });
            
            wrapper.addEventListener('mouseleave', resetMousePosition);
            wrapper.addEventListener('touchend', resetMousePosition);

            // Animasi Lerp (Pergerakan Halus) menggunakan RequestAnimationFrame
            function animateInteraction() {
                // Mendekatkan posisi saat ini ke posisi target secara bertahap
                currentX += (targetX - currentX) * 0.08; 
                currentY += (targetY - currentY) * 0.08;
                
                // Meneruskan posisi ke CSS Variables secara realtime
                wrapper.style.setProperty('--mouseX', `${currentX.toFixed(2)}%`);
                wrapper.style.setProperty('--mouseY', `${currentY.toFixed(2)}%`);
                
                requestAnimationFrame(animateInteraction);
            }
            
            // Mulai loop animasi interaktif
            animateInteraction();
        });
    </script>
